# Character Data

Access comprehensive character data including stats, skills, constellations, and upgrade materials.

## Basic Usage

```ts twoslash
import { Client, Character } from 'genshin-manager'

const client = new Client()
await client.deploy()
// ---cut---
const hutao = new Character(10000046, 90, true, 6)


console.log(hutao.name)       // "Hu Tao"
console.log(hutao.element)    // "Pyro"
console.log(hutao.rarity)     // 5
console.log(hutao.weaponType) // "WEAPON_POLE"
```

## Find Characters

### By Name

```ts twoslash
import { Character } from 'genshin-manager'
// ---cut---
const ids = Character.getCharacterIdByName('Hu Tao')

const hutao = new Character(ids[0])
console.log(hutao.name)
```

### All Characters

```ts twoslash
import { Character } from 'genshin-manager'
// ---cut---
const allIds = Character.allCharacterIds

for (const id of allIds) {
  const char = new Character(id)
  console.log(`${char.name} - ${char.element} - ${char.rarity}★`)
}
```

### Filter by Element

```ts twoslash
import { Character } from 'genshin-manager'
// ---cut---
const pyroCharacters = Character.allCharacterIds
  .map(id => new Character(id))
  .filter(c => c.element === 'Pyro')

console.log(`Pyro characters: ${pyroCharacters.length}`)
```

## Upgrade Materials

### Ascension Materials

```ts twoslash
import { Character, Material } from 'genshin-manager'
// ---cut---
const hutao = new Character(10000046, 80, false, 0)

for (const mat of hutao.ascensionMaterials) {

  const material = new Material(mat.id)
  console.log(`${material.name}: ${mat.count}`)
}

console.log(`Can ascend: ${hutao.isCanAscend}`)
```

### Skill Materials


### Complete Upgrade Plan

```ts twoslash
import { Client, Character, Material } from 'genshin-manager'

const client = new Client()
await client.deploy()
// ---cut---
const hutao = new Character(10000046, 1, false, 0)

const materials = hutao.calculateUpgradeMaterials({
  characterLevel: [1, 90],
  skillLevels: {
    normalAttack: [1, 10],
    elementalSkill: [1, 10],
    elementalBurst: [1, 10],
  }
})


for (const mat of materials) {
  const material = new Material(mat.id)
  console.log(`${material.name}: ${mat.count}`)
}
```

## Traveler (Multi-Element)

The Traveler has different skill sets for each element:

```ts twoslash
import { Character } from 'genshin-manager'
// ---cut---
const travelerId = 10000007 // Female Traveler
const depotIds = Character.getTravelerSkillDepotIds(travelerId)


for (const depotId of depotIds) {
  const traveler = new Character(travelerId, 90, true, 0, depotId)
  console.log(`${traveler.name} - ${traveler.element}`)
}
```

## Multi-Language Support

```ts twoslash
import { Client, Character } from 'genshin-manager'
// ---cut---
const client = new Client({
  defaultLanguage: 'EN',
  downloadLanguages: ['EN', 'JP', 'KR']
})
await client.deploy()

const hutao = new Character(10000046)
console.log(hutao.name) // "Hu Tao"

await client.changeLanguage('JP')
const hutaoJp = new Character(10000046)
console.log(hutaoJp.name) // "胡桃"
```

## Complete Example

```ts twoslash
import { Client, Character, Material } from 'genshin-manager'

async function analyzeCharacter(characterId: number): Promise<void> {
  const client = new Client()
  await client.deploy()

  const char = new Character(characterId, 90, true, 6)

  console.log(`=== ${char.name} ===`)
  console.log(`${char.rarity}★ ${char.element} ${char.weaponType}`)

  // Stats
  for (const stat of char.stats) {
    console.log(`${stat.type}: ${stat.value}`)
  }

  // Skills
  const normalAttack = char.getNormalAttack(10)
  const elementalSkill = char.getElementalSkill(10)
  const elementalBurst = char.getElementalBurst(10)
  console.log(`Normal Attack: ${normalAttack.name}`)
  console.log(`Elemental Skill: ${elementalSkill.name}`)
  console.log(`Elemental Burst: ${elementalBurst.name}`)

  // Profile
  const profile = char.profile
  console.log(`Title: ${profile.title}`)
  console.log(`Constellation: ${profile.constellation}`)
}

await analyzeCharacter(10000046)
```
